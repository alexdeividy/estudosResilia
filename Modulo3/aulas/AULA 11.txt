SELECT * FROM TB_PIZZA
LEFT JOIN TB_PEDIDO_PIZZA
ON TB_PIZZA.ID_PIZZA = TB_PEDIDO_PIZZA.ID_PIZZA;

SELECT A.NOME, B.ID_PEDIDO FROM TB_PIZZA A -- APELIDANDO TABELA PIZZA DE A --
LEFT JOIN TB_PEDIDO_PIZZA B -- APELIDANDO TABELA PEDIDO PIZZA DE B --
ON A.ID_PIZZA = B.ID_PIZZA;

SELECT A.NOME, D.NOME FROM TB_PIZZA A -- SELECIONA ATRIBUTOS NOME, DA TABELA A E B
LEFT JOIN TB_PEDIDO_PIZZA B
ON A.ID_PIZZA = B.ID_PIZZA
LEFT JOIN TB_PEDIDO C
ON C.ID_PEDIDO = B.ID_PEDIDO
LEFT JOIN TB_CLIENTE D
ON D.ID_CLIENTE = C.ID_CLIENTE;
