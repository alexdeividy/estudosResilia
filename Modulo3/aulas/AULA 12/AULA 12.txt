SELECT A.NOME, D.NOME FROM TB_CLIENTE A -- FORMA ERRADA USANDO RIGHT JOIN --
RIGHT JOIN TB_PEDIDO B ON B.ID_CLIENTE = A.ID_CLIENTE
RIGHT JOIN TB_PEDIDO_PIZZA C ON C.ID_PEDIDO = B.ID_PEDIDO
RIGHT JOIN TB_PIZZA D ON D.ID_PIZZA = C.ID_PIZZA;

SELECT D.NOME, A.NOME FROM TB_PIZZA A -- FORMA CORRETA USANDO RIGHT JOIN --
RIGHT JOIN TB_PEDIDO_PIZZA B ON B.ID_PIZZA = A.ID_PIZZA
RIGHT JOIN TB_PEDIDO C ON C.ID_PEDIDO = B.ID_PEDIDO
RIGHT JOIN TB_CLIENTE D ON D.ID_CLIENTE = C.ID_CLIENTE
ORDER BY A.NOME DESC;
