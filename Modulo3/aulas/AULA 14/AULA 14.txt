SELECT * FROM TB_PEDIDO;
SELECT * FROM TB_CLIENTE;


SELECT AVG(PRECO) FROM TB_PEDIDO; -- SELECIONA A MEDIA DE PRECO --

SELECT * FROM TB_PEDIDO WHERE PRECO > (SELECT AVG(PRECO) FROM TB_PEDIDO); -- SELECIONA TODOS OS PEDIDOS ONDE O PREÇO É MAIOR QUE A MÉDIA --

SELECT ID_CLIENTE FROM TB_PEDIDO WHERE TIPO_ENTREGA = 'Retirada no balcão'; -- SELECIONA OS CLIENTES ONDE O TIPO DE ENTREGA É TAL --

-- CONSULTA USANDO JOIN --

SELECT TB_CLIENTE.NOME, TB_PEDIDO.TIPO_ENTREGA FROM TB_PEDIDO
LEFT JOIN TB_CLIENTE
ON TB_CLIENTE.ID_CLIENTE = TB_PEDIDO.ID_CLIENTE
WHERE TIPO_ENTREGA = 'Retirada no balcão';

-- CONSULTA USANDO SUBQUERY --

SELECT NOME FROM TB_CLIENTE WHERE ID_CLIENTE
IN (SELECT ID_CLIENTE FROM TB_PEDIDO WHERE TIPO_ENTREGA = 'Retirada no balcão');
