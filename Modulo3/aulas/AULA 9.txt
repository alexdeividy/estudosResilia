SELECT * FROM TB_CLIENTE WHERE ID_CLIENTE < 5; -- MENOR QUE 5 --
SELECT * FROM TB_CLIENTE WHERE ID_CLIENTE <> 5; -- DIFERENTE DE 5 --

SELECT * FROM TB_PIZZA;
SELECT DISTINCT(CATEGORIA) FROM TB_PIZZA; -- DISTINCT, RETORNA VALORES ÚNICOS NA CONSULTA --
SELECT COUNT (DISTINCT(CATEGORIA)) FROM TB_PIZZA; -- COUNT, CONTA A QUANTIDADE DE REGISTROS --

-- ORDER BY, PERMITE QUE ORDENEMOS NOSSA CONSULTA UTILIZANDO COMO REFERENCIA UMA COLUNA ESPECÍFICA --
SELECT * FROM TB_CLIENTE ORDER BY NOME;
SELECT * FROM TB_CLIENTE ORDER BY NOME DESC; -- DESC, ORDEM DECRESCENTE --

-- GROUP BY, PERMITE QUE FAÇAMOS AGRUPAMENTO PARA OS RESULTADOS DAS CONSULTAS --
SELECT * FROM TB_PEDIDO;
SELECT ID_CLIENTE, SUM(PRECO) FROM TB_PEDIDO GROUP BY ID_CLIENTE;
SELECT ID_CLIENTE, SUM(PRECO) FROM TB_PEDIDO GROUP BY ID_CLIENTE ORDER BY ID_CLIENTE;
SELECT ID_CLIENTE, SUM(PRECO) FROM TB_PEDIDO GROUP BY ID_CLIENTE ORDER BY SUM(PRECO) DESC;

SELECT * FROM TB_PIZZA;

SELECT CATEGORIA, AVG(PRECO) FROM TB_PIZZA GROUP BY CATEGORIA;
SELECT CATEGORIA, COUNT(CATEGORIA) FROM TB_PIZZA GROUP BY CATEGORIA;

-- HAVING, PERMITE APLICAR FILTROS NAS CONSULTAS QUANDO ESTÃO AGRUPADAS --
SELECT CATEGORIA, COUNT(CATEGORIA) FROM TB_PIZZA GROUP BY CATEGORIA HAVING COUNT(CATEGORIA) > 4;
